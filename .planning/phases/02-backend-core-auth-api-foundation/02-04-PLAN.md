**api/src/lib/rbac.ts** — auth guard + role factory:
```typescript
import { Elysia } from 'elysia'
import { jwtPlugin } from './jwt'
import { redis } from '../redis/client'

export type AuthUser = { id: string; role: 'admin' | 'supervisor' | 'analyst' | 'viewer'; jti: string }

/**
 * Reusable auth guard. Use via `.use(authGuard)` in any plugin to require a valid session.
 * Adds `user: AuthUser` to the context.
 */
export const authGuard = new Elysia({ name: 'auth-guard' })
  .use(jwtPlugin)
  .derive({ as: 'scoped' }, async ({ jwt, cookie, set }) => {
    const token = cookie.access_token?.value
    if (!token) {
      set.status = 401
      throw new Error('Unauthorized: no session cookie')
    }
    const payload = await jwt.verify(token)
    if (!payload) {
      set.status = 401
      throw new Error('Unauthorized: invalid token')
    }
    // payload validated against jwtPayloadSchema — has sub/role/jti
    const blacklisted = await redis.get(`session:blacklist:${payload.jti}`)
    if (blacklisted) {
      set.status = 401
      throw new Error('Unauthorized: token revoked')
    }
    return {
      user: {
        id:   payload.sub as string,
        role: payload.role as AuthUser['role'],
        jti:  payload.jti as string,
      } satisfies AuthUser,
    }
  })

/**
 * RBAC factory. Returns a plugin that enforces the given roles.
 * Usage: `tripsPlugin.use(requireRole('admin','supervisor'))`
 */
export function requireRole(...roles: AuthUser['role'][]) {
  return new Elysia({ name: `require-role-${roles.join('-')}` })
    .use(authGuard)
    .onBeforeHandle(({ user, set }) => {
      if (!roles.includes(user.role)) {
        set.status = 403
        throw new Error(`Forbidden: requires role ${roles.join('|')}`)
      }
    })
}
```

Note: `.derive({ as: 'scoped' }, ...)` is critical — without `scoped`, derive only applies to routes defined directly on this plugin, not consumers using `.use(authGuard)`. RESEARCH.md Pattern 3 covers this; Elysia 1.x docs confirm.

```typescript
import bcrypt from 'bcrypt'
import { eq } from 'drizzle-orm'
import { db } from '../../db/client'
import { users, type SelectUser } from '../../db/schema/users'
import { redis } from '../../redis/client'

const BCRYPT_COST = 10

export async function hashPassword(plain: string): Promise<string> {
  return bcrypt.hash(plain, BCRYPT_COST)
}

/**
 * Validates user credentials. Returns the user row on success, null on failure.
 * Uses constant-time comparison via bcrypt.compare.
 * Always runs bcrypt.compare even if user not found, to prevent timing oracle.
 */
const DUMMY_HASH = '$2b$10$abcdefghijklmnopqrstuvOIo2Oh5oP1pP0V8zY5gZ8GlVlZGGGGGG'  // bcrypt-shaped, will never match

export async function validateCredentials(email: string, password: string): Promise<SelectUser | null> {
  const [user] = await db.select().from(users).where(eq(users.email, email)).limit(1)
  // Constant-time defense: always bcrypt.compare even if user missing
  const hashToCompare = user?.passwordHash ?? DUMMY_HASH
  const ok = await bcrypt.compare(password, hashToCompare)
  if (!user || !user.isActive || !ok) return null
  return user
}

/**
 * Adds the jti to Redis blacklist with TTL = remaining lifetime of the JWT.
 * exp is the JWT exp claim (unix seconds).
 */
export async function blacklistJti(jti: string, exp: number): Promise<void> {
  const nowSec = Math.floor(Date.now() / 1000)
  const ttl = exp - nowSec
  if (ttl > 0) {
    await redis.set(`session:blacklist:${jti}`, '1', 'EX', ttl)
  }
}

export async function isJtiBlacklisted(jti: string): Promise<boolean> {
  const v = await redis.get(`session:blacklist:${jti}`)
  return v !== null
}
```

```typescript
import { Elysia, t } from 'elysia'
import { jwtPlugin } from '../../lib/jwt'
import { authGuard } from '../../lib/rbac'
import { redis } from '../../redis/client'
import { logger } from '../../lib/logger'
import { validateCredentials, blacklistJti } from './auth.service'

const LOGIN_RATE_WINDOW_SEC = 60
const LOGIN_RATE_MAX_ATTEMPTS = 10  // per IP per window

const COOKIE_NAME = 'access_token'
const COOKIE_OPTS = {
  httpOnly: true,
  sameSite: 'strict' as const,
  secure: process.env.NODE_ENV === 'production',
  path: '/',
  maxAge: 60 * 60 * 24,  // 24h — matches JWT exp
} satisfies Parameters<NonNullable<unknown>>[0] extends never ? never : Record<string, unknown>

/**
 * Per-IP rate limiter for login. Implemented via Redis INCR + EXPIRE.
 * Returns true if the request should be blocked.
 */
async function isLoginRateLimited(ip: string): Promise<boolean> {
  const key = `ratelimit:login:${ip}`
  const count = await redis.incr(key)
  if (count === 1) {
    await redis.expire(key, LOGIN_RATE_WINDOW_SEC)
  }
  return count > LOGIN_RATE_MAX_ATTEMPTS
}

export const authPlugin = new Elysia({ name: 'auth' })
  .use(jwtPlugin)
  .group('/api/auth', (app) =>
    app
      // ---------- POST /api/auth/login ----------
      .post('/login', async ({ jwt, cookie, body, set, request }) => {
        const ip = request.headers.get('x-forwarded-for') ?? request.headers.get('x-real-ip') ?? 'unknown'
        if (await isLoginRateLimited(ip)) {
          set.status = 429
          return { error: 'Too many login attempts. Try again later.' }
        }
        const user = await validateCredentials(body.email, body.password)
        if (!user) {
          set.status = 401
          logger.warn({ email: body.email, ip }, 'login failed')
          return { error: 'Invalid credentials' }
        }
        const jti = crypto.randomUUID()
        const token = await jwt.sign({ sub: user.id, role: user.role as 'admin'|'supervisor'|'analyst'|'viewer', jti })
        cookie[COOKIE_NAME].set({ value: token, ...COOKIE_OPTS })
        logger.info({ userId: user.id, email: user.email }, 'login success')
        return { user: { id: user.id, name: user.name, email: user.email, role: user.role } }
      }, {
        body: t.Object({
          email:    t.String({ format: 'email' }),
          password: t.String({ minLength: 6 }),
        }),
        detail: { tags: ['auth'], summary: 'Login with email + password' },
      })

      // ---------- POST /api/auth/logout ----------
      .post('/logout', async ({ jwt, cookie, set }) => {
        const token = cookie[COOKIE_NAME]?.value
        if (token) {
          const payload = await jwt.verify(token)
          if (payload && payload.jti && payload.exp) {
            await blacklistJti(payload.jti as string, payload.exp as number)
          }
        }
        cookie[COOKIE_NAME].remove()
        set.status = 204
        return ''
      }, {
        detail: { tags: ['auth'], summary: 'Logout — blacklist current token' },
      })

      // ---------- GET /api/auth/me ----------
      .use(authGuard)  // attaches `user` to context
      .get('/me', ({ user }) => {
        return { user: { id: user.id, role: user.role } }
      }, {
        detail: { tags: ['auth'], summary: 'Current authenticated user' },
      })

      // ---------- POST /api/auth/refresh ----------
      .post('/refresh', async ({ jwt, cookie, user, set }) => {
        // Issue new token, blacklist the previous jti
        await blacklistJti(user.jti, Math.floor(Date.now() / 1000) + 60 * 60 * 24)  // worst-case: full 24h
        const newJti = crypto.randomUUID()
        const newToken = await jwt.sign({ sub: user.id, role: user.role, jti: newJti })
        cookie[COOKIE_NAME].set({ value: newToken, ...COOKIE_OPTS })
        return { ok: true }
      }, {
        detail: { tags: ['auth'], summary: 'Issue a fresh token, invalidating the previous one' },
      })
  )
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → /api/auth/login | Public unauthenticated entry point — strongest validation here |
| browser → /api/auth/logout, /me, /refresh | Cookie-bearing requests; cookie value is the trust anchor |
| API → Redis (blacklist) | Trusted internal service; all keys namespaced |
| API → Postgres (users) | Trusted internal service; password hashes only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-04-01 | Spoofing | Login bypass via SQL injection in email | mitigate | Drizzle `eq(users.email, email)` is parameterized; raw SQL not used. |
| T-02-04-02 | Spoofing | Login bypass via timing attack (existence oracle) | mitigate | `validateCredentials` uses DUMMY_HASH constant-time `bcrypt.compare` even when user missing. |
| T-02-04-03 | Tampering | JWT forgery | mitigate | HS256 with secret >= 32 chars enforced at startup; production must inject 64-byte hex. |
| T-02-04-04 | Information Disclosure | JWT theft via XSS | mitigate | HttpOnly cookie — JS cannot read `access_token`. SameSite=Strict prevents cross-site exfil. |
| T-02-04-05 | Repudiation | User can repudiate logout (token still valid) | mitigate | Logout adds jti to Redis `session:blacklist:{jti}` with TTL = exp; authGuard rejects blacklisted jti. |
| T-02-04-06 | Elevation of Privilege | Brute-force password guessing | mitigate | Per-IP rate limit on /login: 10 attempts / 60s window via Redis counter. ASVS V2.2.1 satisfied. |
| T-02-04-07 | Elevation of Privilege | RBAC bypass — user escalates role by editing JWT | mitigate | HS256 signature prevents tampering; `requireRole()` checks role from verified payload, not from request body. |
| T-02-04-08 | Tampering | CSRF via cross-site form post | mitigate | `SameSite=Strict` cookie — browser refuses to send on cross-origin requests. CORS `origin: FRONTEND_URL` (Plan 06) blocks unrelated origins. ASVS V3.5.1 satisfied for app-internal use. |
| T-02-04-09 | Information Disclosure | Password hash leakage in logs | mitigate | logger.info call passes `email` only (not passwordHash); user object response strips passwordHash via explicit field selection. |
| T-02-04-10 | Denial of Service | Token-validation Redis call latency on every request | accept | Single GET per request; <2ms typical for ioredis local. Future: in-process LRU cache (deferred to Phase 6). |
</threat_model>
