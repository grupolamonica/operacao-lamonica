    **File 1 — api/src/modules/thresholds/thresholds.service.ts** (copy RESEARCH.md lines 1283-1313):
    ```typescript
    import { db } from '../../db/client'
    import { alertThresholds } from '../../db/schema/alert-thresholds'

    let cache: Record<string, number> | null = null
    let cacheExpiry = 0
    const TTL_MS = 60_000  // 60 seconds, matches D-19

    export async function getThresholds(): Promise<Record<string, number>> {
      if (cache && Date.now() < cacheExpiry) return cache
      const rows = await db.select().from(alertThresholds)
      cache = Object.fromEntries(rows.map(r => [r.type, r.value]))
      cacheExpiry = Date.now() + TTL_MS
      return cache
    }

    export function invalidateThresholdsCache(): void {
      cache = null
      cacheExpiry = 0
    }

    export async function updateThreshold(type: string, value: number, updatedBy: string) {
      await db.insert(alertThresholds)
        .values({ type, value, updatedBy })
        .onConflictDoUpdate({
          target: alertThresholds.type,
          set:    { value, updatedBy, updatedAt: new Date() },
        })
      invalidateThresholdsCache()
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Authenticated client → /api/users/* (admin) | Server validates role=admin via requireRole; password input sent over HTTPS, hashed before persist |
| Authenticated user → /api/users/me/notification-preferences | Self-only modification; user.id from JWT, never from client body |
| Authenticated client → /api/thresholds (read) | Any role |
| Admin client → /api/thresholds/:type (write) | requireRole admin |
| Admin client → /api/gps-providers/* (write) | requireRole admin; apiKey stored in cleartext as STUB only — no real integration this phase |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06.03-01 | Elevation of privilege | /api/users/* CRUD | mitigate | requireRole('admin') enforced server-side; 403 if non-admin (ASVS V4.1.1) |
| T-06.03-02 | Information disclosure | listUsers response | mitigate | Projection excludes passwordHash; verify no row includes passwordHash field |
