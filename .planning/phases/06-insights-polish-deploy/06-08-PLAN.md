    ```yaml
    name: Deploy
    on:
      push:
        branches: [main]

    jobs:
      detect:
        runs-on: ubuntu-latest
        outputs:
          backend:  ${{ steps.filter.outputs.backend }}
          frontend: ${{ steps.filter.outputs.frontend }}
        steps:
          - uses: actions/checkout@v4
          - id: filter
            uses: dorny/paths-filter@v3
            with:
              filters: |
                backend:
                  - 'api/**'
                  - 'railway.json'
                frontend:
                  - 'torre-de-controle/**'

      deploy-backend:
        needs: detect
        if: needs.detect.outputs.backend == 'true'
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: oven-sh/setup-bun@v2
            with: { bun-version: 1.3.13 }
          - name: Install
            run: cd api && bun install --frozen-lockfile
          # CRITICAL: --strict requires interactive confirmation on destructive ops.
          # In CI (no TTY), this means DESTRUCTIVE schema changes HANG until timeout — protection per D-37 risk acceptance.
          # Non-destructive additions proceed automatically.
          - name: Drizzle schema push (production)
            env:
              DATABASE_URL: ${{ secrets.DATABASE_URL_PROD }}
            run: cd api && bunx drizzle-kit push --strict --verbose
          - name: Install Railway CLI
            run: npm install -g @railway/cli
          - name: Deploy to Railway
            env:
              RAILWAY_TOKEN: ${{ secrets.RAILWAY_TOKEN }}
            run: railway up --service=${{ secrets.RAILWAY_SERVICE_ID }} --detach

      deploy-frontend:
        needs: detect
        if: needs.detect.outputs.frontend == 'true'
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: '20'
              cache: 'npm'
              cache-dependency-path: torre-de-controle/package-lock.json
          - name: Install
            run: cd torre-de-controle && npm ci
          - name: Build with Sentry source maps upload
            env:
              VITE_API_URL:           ${{ secrets.VITE_API_URL }}
              VITE_SENTRY_DSN:        ${{ secrets.VITE_SENTRY_DSN }}
              VITE_VAPID_PUBLIC_KEY:  ${{ secrets.VITE_VAPID_PUBLIC_KEY }}
              SENTRY_AUTH_TOKEN:      ${{ secrets.SENTRY_AUTH_TOKEN }}
              SENTRY_ORG:             ${{ secrets.SENTRY_ORG }}
              SENTRY_PROJECT:         torre-frontend
            run: cd torre-de-controle && npm run build
          - name: Deploy to Cloudflare Pages
            uses: cloudflare/wrangler-action@v3
            with:
              apiToken:    ${{ secrets.CLOUDFLARE_API_TOKEN }}
              accountId:   ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
              command:     pages deploy torre-de-controle/dist --project-name=torre-de-controle --branch=main
              gitHubToken: ${{ secrets.GITHUB_TOKEN }}
    ```

    ### A) Railway setup
    1. Visit https://railway.app and sign up/log in.
    2. Click "New Project" → "Empty Project".
    3. Project name: `torre-de-controle`.
    4. Click "New" → "Database" → "Add PostgreSQL". After provisioning, open PostgreSQL service → Query → run:
       ```sql
       CREATE EXTENSION IF NOT EXISTS postgis;
       ```
    5. Click "New" → "Database" → "Add Redis".
    6. Click "New" → "GitHub Repo" → select this repo. Service detects `railway.json`, uses Bun start command.
    7. In API service → Variables tab, set ALL backend env vars (see user_setup railway section above). Critical:
       - `DATABASE_URL` and `REDIS_URL` are auto-injected by Railway from plugins (do NOT set manually).
       - `JWT_SECRET`: generate via `openssl rand -hex 32` (32-byte hex string).
       - `VAPID_PUBLIC_KEY` + `VAPID_PRIVATE_KEY` + `VAPID_SUBJECT`: generate locally (Step D below), persist here.
       - `FRONTEND_URL`: temporarily `https://torre-de-controle.pages.dev` (Cloudflare default subdomain — confirm after CF setup).
       - `SENTRY_DSN`: copy from torre-api Sentry project (Step C).
       - `TELEMETRY_API_KEY`: rotate from dev key — generate new random string.
    8. Copy from Railway:
       - Service ID (Service → Settings → Service ID) → save as `RAILWAY_SERVICE_ID` GH Secret
       - Project token (Account → Tokens) → save as `RAILWAY_TOKEN` GH Secret
       - DATABASE_URL (PostgreSQL → Variables → DATABASE_URL) → save as `DATABASE_URL_PROD` GH Secret

    ### B) Cloudflare Pages setup
    1. Visit https://dash.cloudflare.com.
    2. Workers & Pages → Create application → Pages → Connect to Git.
    3. Repository: this repo. Project name: `torre-de-controle`.
    4. Build settings:
       - Framework preset: None
       - Build command: `cd torre-de-controle && npm ci && npm run build`
       - Build output: `torre-de-controle/dist`
       - Root directory: `/` (leave blank)
    5. Skip env vars in CF dashboard — GH Actions injects them at build time (`VITE_*` vars).
    6. Initial deploy will fail (first run) — that's OK; subsequent GH Actions push deploys correctly.
    7. Copy from Cloudflare:
       - Account ID (right sidebar) → save as `CLOUDFLARE_ACCOUNT_ID` GH Secret
       - Create API Token (My Profile → API Tokens → Create Token; permissions: Account Cloudflare Pages: Edit + Account Settings: Read) → save as `CLOUDFLARE_API_TOKEN` GH Secret

    ### C) Sentry setup
    1. Visit https://sentry.io and sign up/log in.
    2. Create organization (or use existing).
    3. Create 2 projects:
