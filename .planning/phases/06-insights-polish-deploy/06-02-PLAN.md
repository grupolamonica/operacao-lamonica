    **(1) `getSlaHistory(range: Range = '30d')`**:
    SQL from RESEARCH.md lines 263-281 (exact):
    ```
    SELECT DATE(window_end) AS date, COUNT(*) AS total,
           COUNT(*) FILTER (WHERE sla_status = 'no_prazo') AS on_time
    FROM trips
    WHERE status = 'completed' AND window_end >= NOW() - INTERVAL '1 day' * ${days}
    GROUP BY DATE(window_end) ORDER BY date ASC
    ```
    Map rows to `{ date: string, total: number, onTime: number, sla: number }` where sla = onTime/total (0 if total=0).

    **(2) `getDriversRanking(range: Range = '30d', limit: number = 10)`**:
    Aggregate trips JOIN drivers WHERE trips.status='completed' AND trips.window_end >= NOW() - INTERVAL... GROUP BY driver. Compute slaPercent = COUNT(*) FILTER (WHERE sla_status='no_prazo') / COUNT(*), avgDelayMin = AVG(EXTRACT(EPOCH FROM (arrived_at - window_end))/60) FILTER (WHERE arrived_at > window_end). ORDER BY operationalScore DESC LIMIT $limit. Map to `{ driverId, name, code, score, slaPercent, avgDelayMin, totalTrips }`. Reference RESEARCH.md lines 184-185 (endpoint contract).

    **(3) `getProblematicRoutes(range: Range = '30d')`**:
    Aggregate alerts JOIN trips JOIN routes WHERE alerts.occurred_at >= NOW() - INTERVAL... GROUP BY route. Compute alerts count, avgDelay (sum of trips with route's alert -> delay_minutes mean), slaPercent (trips with route completed on_time/total). ORDER BY alerts DESC LIMIT 20. Map to `{ routeId, code, name, alerts, avgDelay, slaPercent }`. Reference RESEARCH.md lines 187-188.

    **(4) `getAlertsDistribution(range: Range = '30d')`**:
    SQL: `SELECT type, COUNT(*) AS count FROM alerts WHERE occurred_at >= NOW() - INTERVAL '1 day' * ${days} GROUP BY type ORDER BY count DESC`. Map to `{ type: string, count: number }`. Reference RESEARCH.md lines 190-191.

    ```typescript
    export const BOM = '﻿'  // U+FEFF — UTF-8 BOM, Excel BR opens correctly

    export function formatCsvRow(values: Array<string | number | null | undefined>): string {
      return values.map(v => {
        if (v === null || v === undefined) return ''
        const s = String(v)
        if (s.includes(';') || s.includes('"') || s.includes('\n')) {
          return `"${s.replace(/"/g, '""')}"`
        }
        return s
      }).join(';')
    }

    export function dateStamp(): string {
      const d = new Date()
      const pad = (n: number) => String(n).padStart(2, '0')
      return `${d.getFullYear()}-${pad(d.getMonth() + 1)}-${pad(d.getDate())}_${pad(d.getHours())}${pad(d.getMinutes())}`
    }
    ```

    ```typescript
    const TRIPS_HEADER = ['Código','Motorista','Cliente','Origem','Destino','Janela Início','Janela Fim','ETA','Status','SLA','Progresso %']

    export function streamTripsCsv(filters: { status?: string; slaStatus?: string; clientName?: string; search?: string }): ReadableStream<Uint8Array> {
      const encoder = new TextEncoder()
      return new ReadableStream({
        async start(controller) {
          controller.enqueue(encoder.encode(BOM + TRIPS_HEADER.join(';') + '\n'))

          const conditions = []
          if (filters.status)    conditions.push(eq(trips.status, filters.status as any))
          if (filters.slaStatus) conditions.push(eq(trips.slaStatus, filters.slaStatus as any))
          // ... clientName/search omitted in v1 — apply if straightforward
          const where = conditions.length ? and(...conditions) : undefined

          const q = where ? db.select().from(trips).where(where).limit(50000) : db.select().from(trips).limit(50000)
          for await (const row of q) {
            const line = formatCsvRow([
              row.code, row.driverId, row.clientId,
              row.origin, row.destination,
              row.windowStart?.toISOString(), row.windowEnd?.toISOString(),
              row.eta?.toISOString(), row.status, row.slaStatus, row.progressPct,
            ])
            controller.enqueue(encoder.encode(line + '\n'))
          }
          controller.close()
        },
      })
    }
    ```
