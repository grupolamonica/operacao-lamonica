    **File 1 — torre-de-controle/src/hooks/useExportCsv.ts**:
    ```typescript
    export type ExportEntity = 'viagens' | 'alertas' | 'tratativas' | 'motoristas'

    export function useExportCsv() {
      return (entity: ExportEntity, filters: Record<string, any> = {}) => {
        const apiUrl = (import.meta.env.VITE_API_URL as string | undefined) ?? 'http://localhost:3000'
        const params = Object.entries(filters)
          .filter(([_, v]) => v !== undefined && v !== '' && v !== null)
          .reduce((acc, [k, v]) => ({ ...acc, [k]: String(v) }), {})
        const qs = new URLSearchParams(params).toString()
        // window.location.href: browser carries HttpOnly cookie automatically (D-09)
        // Note: changes page location but browser handles content-disposition: attachment, so user stays on current page
        window.location.href = `${apiUrl}/api/exports/${entity}.csv${qs ? '?' + qs : ''}`
      }
    }
    ```
