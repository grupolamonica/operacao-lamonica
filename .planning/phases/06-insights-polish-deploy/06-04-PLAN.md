    ```typescript
    import webpush from 'web-push'
    import { eq } from 'drizzle-orm'
    import { db } from '../../db/client'
    import { pushSubscriptions } from '../../db/schema/push-subscriptions'
    import { logger } from '../../lib/logger'
    import { isVapidConfigured, vapidPublicKey } from '../../lib/vapid'

    export function getVapidPublicKey(): string | null {
      return vapidPublicKey ?? null
    }

    export async function subscribe(userId: string, input: { endpoint: string; keys: { p256dh: string; auth: string } }) {
      await db.insert(pushSubscriptions).values({
        userId,
        endpoint: input.endpoint,
        p256dh:   input.keys.p256dh,
        auth:     input.keys.auth,
      }).onConflictDoNothing({ target: pushSubscriptions.endpoint })
      logger.info({ userId, endpoint: input.endpoint.slice(0, 40) + '...' }, 'push subscription registered')
    }

    export async function unsubscribe(userId: string, endpoint: string) {
      await db.delete(pushSubscriptions).where(eq(pushSubscriptions.endpoint, endpoint))
      logger.info({ userId }, 'push subscription removed')
    }

    export async function sendToUser(userId: string, payload: { title: string; body: string; url: string }) {
      if (!isVapidConfigured) {
        logger.warn({ userId }, 'sendToUser called but VAPID not configured — skipping')
        return
      }
      const subs = await db.select().from(pushSubscriptions).where(eq(pushSubscriptions.userId, userId))
      await Promise.allSettled(subs.map(async s => {
        try {
          await webpush.sendNotification(
            { endpoint: s.endpoint, keys: { p256dh: s.p256dh, auth: s.auth } },
            JSON.stringify(payload),
            { TTL: 60 },
          )
        } catch (e: any) {
          if (e.statusCode === 410 || e.statusCode === 404) {
            // Subscription expired — cleanup
            await db.delete(pushSubscriptions).where(eq(pushSubscriptions.endpoint, s.endpoint))
            logger.info({ endpoint: s.endpoint.slice(0, 40) + '...', statusCode: e.statusCode }, 'cleaned up expired subscription')
          } else {
            logger.error({ error: e.message, statusCode: e.statusCode, endpoint: s.endpoint.slice(0, 40) + '...' }, 'push send failed')
          }
        }
      }))
    }
    ```

    **File 2 — api/src/modules/push/push.dispatcher.ts** (RESEARCH lines 455-470):
    ```typescript
    import { sql } from 'drizzle-orm'
    import { db } from '../../db/client'
    import { logger } from '../../lib/logger'
    import { sendToUser } from './push.service'

    export type AlertPushPayload = {
      id: string
      title: string
      description: string
      severity: 'critico' | 'medio' | 'baixo'
    }

    export async function dispatchAlertPush(alert: AlertPushPayload) {
      try {
        // SQL: notification_preferences->>'critico' = 'true' (JSONB ->> returns text)
        const users = await db.execute(sql`
          SELECT id FROM users
          WHERE is_active = true
            AND notification_preferences->>${alert.severity} = 'true'
        `) as Array<{ id: string }>

        if (users.length === 0) return

        await Promise.allSettled(users.map(u =>
          sendToUser(u.id, {
            title: `⚠ ${alert.title}`,
            body:  alert.description ?? '',
            url:   `/alertas/${alert.id}`,
          })
        ))
        logger.info({ alertId: alert.id, severity: alert.severity, recipientCount: users.length }, 'alert push dispatched')
      } catch (e: any) {
        logger.error({ error: e.message, alertId: alert.id }, 'dispatchAlertPush failed')
      }
    }
    ```

    // GET vapid public key — no auth (public key not secret)
    const publicKeyPlugin = new Elysia({ name: 'push-public-key' })
      .get('/api/push/vapid-public-key', () => {
        const key = getVapidPublicKey()
        return { publicKey: key }
      }, {
        detail: { tags: ['push'], summary: 'Get VAPID public key for client subscribe' },
      })

    // Subscribe/Unsubscribe — authenticated users
    const authedPlugin = new Elysia({ name: 'push-authed' })
      .use(authGuard)
      .group('/api/push', (app) =>
        app
          .post('/subscribe', async ({ body, user, set }) => {
            await subscribe(user.id, body)
            set.status = 201
            return { ok: true }
          }, {
            body: t.Object({
              endpoint: t.String({ minLength: 1 }),
              keys: t.Object({
                p256dh: t.String({ minLength: 1 }),
                auth:   t.String({ minLength: 1 }),
              }),
            }),
            detail: { tags: ['push'], summary: 'Register push subscription' },
          })
          .post('/unsubscribe', async ({ body, user, set }) => {
            await unsubscribe(user.id, body.endpoint)
            set.status = 204; return ''
          }, {
            body: t.Object({ endpoint: t.String({ minLength: 1 }) }),
            detail: { tags: ['push'], summary: 'Remove push subscription' },
          })
      )

    export const pushPlugin = new Elysia({ name: 'push' })
      .use(publicKeyPlugin)
      .use(authedPlugin)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Alert engine → push.dispatcher | Internal call; payload trusted (built from DB record) |
| push.service → external Push services (FCM, Mozilla autopush, Apple) | VAPID-signed; subscription endpoints stored in DB |
| Any client → GET /api/push/vapid-public-key | NO auth — public key is not secret per VAPID spec |
| Authenticated client → POST /api/push/subscribe | userId from JWT, NOT from client body |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06.04-01 | Spoofing | /api/push/subscribe with arbitrary userId | mitigate | user.id derived from authGuard JWT (server-side), body only carries subscription data — userId never trusted from client |
| T-06.04-02 | DoS | Push subscription spam (one user creates 1000s of subscriptions) | mitigate | UNIQUE constraint on endpoint dedupes; onConflictDoNothing. Optional follow-up: rate-limit /api/push/subscribe to 10/min/user (out of scope this phase) |
| T-06.04-03 | DoS | Alert engine blocks waiting for push completion | mitigate | dispatchAlertPush(...).catch(...) fire-and-forget; await is INSIDE dispatcher's own Promise.allSettled but caller (alert-inline.ts) does not await |
| T-06.04-04 | Information disclosure | Push payload contains alert title with PII | accept | Operators need actionable info; payload sent over HTTPS via push service. Sentry beforeSend scrubs identifiers from error reports if a push fails (Plan 01 scrub) |
| T-06.04-05 | Spoofing | VAPID private key exposure | mitigate | api/src/lib/vapid.ts reads from process.env only; .env.example shipped with empty values; real values in Railway secret store (per D-44) |
| T-06.04-06 | Tampering | dispatchAlertPush JSONB injection via severity | mitigate | severity typed as 'critico'|'medio'|'baixo' literal union; SQL parameter interpolated via Drizzle sql template (parameterized, not concatenated); jsonb->>${severity} forces text comparison |
| T-06.04-07 | Resource exhaustion | sendToUser fans out to thousands of subs | accept | MVP user count <50; if scaled, add batching / queue (BullMQ deferred per CONTEXT D-09). Operationally bounded by team size |
| T-06.04-08 | Information disclosure | Error logs leak push endpoint URLs | mitigate | logger.info/error truncates endpoint to first 40 chars + '...' to avoid full URL in logs (push endpoint contains auth identifier) |
